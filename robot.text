function afficherResultat(score){
    const zoneScore = document.querySelector(".zoneScore span")
    return zoneScore.innerText = `${score}/${3}`
}

function affichageProposition(proposition){
    let zoneProposition = document.querySelector(" .zoneProposition")
        zoneProposition.innerHTML = proposition
        
}

function lencerLejeux(){
    const button = document.querySelector("button")
    let score = 0
    let i= 0
    if(inputEcriture.value === listesMots[i]) {
        score++
    }
    i++
    afficherResultat(score, i)
    affichageProposition(listesMots[i])

    button.addEventListener("click",()=>{

        inputEcriture.value = " "

        i++

        if(listesMots[i] === undefined) {
            affichageProposition("le jeux est fini")
           return button.disabled= true

        } else{

            affichageProposition(listesMots[i])
        }

        afficherResultat(score, i)
    })

    
}
